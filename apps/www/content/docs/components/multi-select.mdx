## Usage

```tsx
import { MultiSelect } from "@/components/ui/multi-select"
```

```tsx
<MultiSelect
  options={[
    { value: "React", label: "React" },
    { value: "Vue", label: "Vue" },
    { value: "Svelte", label: "Svelte" },
    { value: "Angular", label: "Angular" },
  ]}
  selected={selected}
  onChange={setSelected}
  className="sm:w-[300px]"
  placeholder="Select a framework"
/>
```
