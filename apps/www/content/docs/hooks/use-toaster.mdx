## Installation

1. Install [Toast](/docs/) component from shadcn/ui.

2. Copy and paste the following code into your project.

<ComponentSource src="/hooks/use-toaster.ts" />

<Callout>
  This is the `useToaster()` hook. You can place it in a file at
  `hooks/use-toaster.ts`.
</Callout>

## Usage

```tsx
import { useToaster } from "@/hooks/use-toaster"

import {
  Toast,
  ToastClose,
  ToastProvider,
  ToastViewport,
} from "@/components/ui/toast"

const Toaster = () => {
  const { toasts } = useToaster()

  return (
    <ToastProvider>
      {toasts.map(({ id, children, ...props }) => (
        <Toast key={id}>
          {children}
          <ToastClose />
        </Toast>
      ))}

      <ToastViewport position="bottom-center" />
    </ToastProvider>
  )
}
```

```tsx
import { toast } from "@/hooks/use-toaster"

import { Button } from "@/components/ui/button"
```

```tsx
<Button
  onClick={() =>
    toast({
      content: (
        <>
          <ToastTitle>New toast</ToastTitle>
          <ToastDescription>Description of the toast</ToastDescription>
        </>
      ),
    })
  }
>
  Toaster
</Button>
```
